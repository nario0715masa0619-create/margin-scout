<template>
  <div class="candidate-list-container">
    <h1>📊 候補一覧</h1>

    <!-- フィルター・ソートコントロール -->
    <div class="controls">
      <input
        v-model="searchText"
        type="text"
        placeholder="🔍 商品名で検索"
        class="search-input"
      />
      <select v-model="sortBy" class="sort-select">
        <option value="profit_desc">利益が多い順</option>
        <option value="profit_asc">利益が少ない順</option>
        <option value="match_score">マッチスコア順</option>
      </select>
      <button @click="selectAll" class="btn btn-secondary">全選択</button>
      <button @click="clearSelection" class="btn btn-secondary">選択解除</button>
    </div>

    <!-- 候補テーブル -->
    <div v-if="filteredCandidates.length > 0" class="table-wrapper">
      <table class="candidates-table">
        <thead>
          <tr>
            <th width="40"><input v-model="selectAllCheckbox" type="checkbox" @change="toggleSelectAll" /></th>
            <th>商品名</th>
            <th>取得元</th>
            <th>仕入価格</th>
            <th>eBay価格</th>
            <th>利益</th>
            <th>利益率</th>
            <th>スコア</th>
          </tr>
        </thead>
        <tbody>
          <tr
            v-for="candidate in filteredCandidates"
            :key="candidate.candidateId"
            :class="['candidate-row', candidate.profitJpy < 0 ? 'loss' : 'profit']"
          >
            <td>
              <input
                :checked="store.selectedCandidates.has(candidate.candidateId)"
                @change="store.toggleSelectedCandidate(candidate.candidateId)"
                type="checkbox"
              />
            </td>
            <td>
              <router-link
                :to="{ name: 'CandidateDetail', params: { jobId: route.params.jobId, candidateId: candidate.candidateId } }"
                class="product-link"
              >
                {{ candidate.productName }}
              </router-link>
            </td>
            <td>{{ candidate.sourceChannel }}</td>
            <td class="text-right">¥{{ formatNumber(candidate.sourcePrice) }}</td>
            <td class="text-right">${{ candidate.ebayPrice.toFixed(2) }}</td>
            <td :class="['text-right', candidate.profitJpy >= 0 ? 'profit-value' : 'loss-value']">
              ¥{{ formatNumber(candidate.profitJpy) }}
            </td>
            <td :class="['text-right', candidate.profitMarginPct >= 0 ? 'profit-value' : 'loss-value']">
              {{ candidate.profitMarginPct.toFixed(1) }}%
            </td>
            <td class="text-right">{{ candidate.matchScore.toFixed(3) }}</td>
          </tr>
        </tbody>
      </table>
    </div>

    <!-- データなし表示 -->
    <div v-else class="empty-state">
      <p>📭 該当する候補がありません</p>
    </div>

    <!-- CSV出力ボタン -->
    <div class="footer-actions">
      <button
        @click="goToExport"
        :disabled="store.selectedCandidates.size === 0"
        class="btn btn-primary"
      >
        📥 選択候補をCSV出力 ({{ store.selectedCandidates.size }} 件)
      </button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useResearchStore } from '../stores/research'
import { researchAPI } from '../services/api'
import type { Candidate } from '../stores/research'

const route = useRoute()
const router = useRouter()
const store = useResearchStore()

const searchText = ref('')
const sortBy = ref('profit_desc')
const selectAllCheckbox = ref(false)
const candidates = ref<Candidate[]>([])
const isLoading = ref(false)

const filteredCandidates = computed(() => {
  let filtered = candidates.value.filter(c =>
    c.productName.toLowerCase().includes(searchText.value.toLowerCase())
  )

  if (sortBy.value === 'profit_desc') {
    filtered.sort((a, b) => b.profitJpy - a.profitJpy)
  } else if (sortBy.value === 'profit_asc') {
    filtered.sort((a, b) => a.profitJpy - b.profitJpy)
  } else if (sortBy.value === 'match_score') {
    filtered.sort((a, b) => b.matchScore - a.matchScore)
  }

  return filtered
})

const selectAll = () => {
  filteredCandidates.value.forEach(c => {
    if (!store.selectedCandidates.has(c.candidateId)) {
      store.toggleSelectedCandidate(c.candidateId)
    }
  })
}

const clearSelection = () => {
  store.clearSelection()
}

const toggleSelectAll = () => {
  if (selectAllCheckbox.value) {
    selectAll()
  } else {
    clearSelection()
  }
}

const goToExport = () => {
  router.push({ name: 'CsvExport', params: { jobId: route.params.jobId } })
}

const formatNumber = (num: number) => {
  return num.toLocaleString('ja-JP')
}

const fetchCandidates = async () => {
  try {
    isLoading.value = true
    const jobId = route.params.jobId as string
    const data = await researchAPI.getCandidates(jobId, 100, 0)

    candidates.value = data.items || []
    store.setCandidatesList(candidates.value)
  } catch (err: any) {
    console.error('Failed to fetch candidates:', err)
  } finally {
    isLoading.value = false
  }
}

onMounted(() => {
  fetchCandidates()
})
</script>
